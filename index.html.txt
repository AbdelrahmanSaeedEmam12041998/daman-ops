            <input type="text" id="merchantName">

            <select id="providerStatus">
                <option value="Pending">Pending</option>

    <script>
        // ضع الرابط الخاص بك هنا
        const scriptURL = 'رابط_جوجل_سكريبت_هنا';
        const form = document.getElementById('disputeForm');

        form.addEventListener('submit', e => {
            e.preventDefault();
            const btn = document.getElementById('submitBtn');
            btn.disabled = true;
            btn.innerText = "جاري الحفظ...";

            const formData = {
                providerOpNum: document.getElementById('providerOpNum').value,
                opNum: document.getElementById('opNum').value,
                amount: document.getElementById('amount').value,
                merchantName: document.getElementById('merchantName').value,
                providerStatus: document.getElementById('providerStatus').value,
                extraInfo: "Web App Entry"
            };

            fetch(scriptURL, { method: 'POST', body: JSON.stringify(formData) })
            .then(response => {
                alert('تم التسجيل بنجاح في الشيت!');
                form.reset();
                btn.disabled = false;
                btn.innerText = "حفظ في Google Sheets";
            })
            .catch(error => {
                alert('خطأ في الإرسال!');
                btn.disabled = false;
            });
        });
    </script>
